!============================================
! ROUTER LANTAI 4 - Gedung ARA Tech
! Departemen: Customer Service, Keuangan, Legal
! Network Aggregation: 192.168.16.192/24
!============================================

enable
configure terminal

hostname Router-Lantai4

! Interface ke Router Gedung ARA
interface FastEthernet0/0
 description Connection to Router-Gedung-ARA
 ip address 192.168.20.14 255.255.255.252
 no shutdown

! Interface ke Switch Customer Service (VLAN 10)
interface FastEthernet1/0.10
 description Customer Service Department
 encapsulation dot1Q 10
 ip address 192.168.16.193 255.255.255.192
 no shutdown

! Interface ke Switch Keuangan (VLAN 20)
interface FastEthernet1/0.20
 description Finance Department
 encapsulation dot1Q 20
 ip address 192.168.17.65 255.255.255.192
 no shutdown

! Interface ke Switch Legal (VLAN 30)
interface FastEthernet1/0.30
 description Legal Department
 encapsulation dot1Q 30
 ip address 192.168.18.1 255.255.255.224
 no shutdown

! Enable parent interface
interface FastEthernet1/0
 description Trunk to Switches Lantai 4
 no shutdown

!============================================
! EIGRP Configuration
!============================================

router eigrp 100
 network 192.168.20.12 0.0.0.3
 network 192.168.16.192 0.0.0.63
 network 192.168.17.64 0.0.0.63
 network 192.168.18.0 0.0.0.31
 passive-interface FastEthernet1/0.10
 passive-interface FastEthernet1/0.20
 passive-interface FastEthernet1/0.30
 no auto-summary

!============================================
! DHCP Pools
!============================================

! Customer Service DHCP
ip dhcp excluded-address 192.168.16.193 192.168.16.200
ip dhcp pool CS-Pool
 network 192.168.16.192 255.255.255.192
 default-router 192.168.16.193
 dns-server 8.8.8.8

! Keuangan DHCP
ip dhcp excluded-address 192.168.17.65 192.168.17.72
ip dhcp pool Keuangan-Pool
 network 192.168.17.64 255.255.255.192
 default-router 192.168.17.65
 dns-server 8.8.8.8

! Legal DHCP
ip dhcp excluded-address 192.168.18.1 192.168.18.7
ip dhcp pool Legal-Pool
 network 192.168.18.0 255.255.255.224
 default-router 192.168.18.1
 dns-server 8.8.8.8

!============================================
! Basic Configuration
!============================================

no ip domain-lookup
service password-encryption

line con 0
 password cisco
 login
 logging synchronous

line vty 0 4
 password cisco
 login

enable secret class

banner motd #
========================================
 ROUTER LANTAI 4 - ARA Tech
========================================
#

end
write memory
