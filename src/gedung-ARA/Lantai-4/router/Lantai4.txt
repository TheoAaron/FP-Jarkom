!============================================
! ROUTER LANTAI 4 - Gedung ARA Tech
! Departemen: Customer Service, Keuangan, Legal
! Network Aggregation: 192.168.16.192/24
!============================================

enable
configure terminal

hostname Router-Lantai4

! Interface ke Router Gedung ARA
interface FastEthernet0/0
 description Connection to Router-Gedung-ARA
 ip address 192.168.10.130 255.255.255.252
 no shutdown

! Interface ke Switch Customer Service (VLAN 10)
interface FastEthernet1/0.10
 description Customer Service Department
 encapsulation dot1Q 10
 ip address 192.168.7.129 255.255.255.192
 no shutdown

! Interface ke Switch Keuangan (VLAN 20)
interface FastEthernet1/0.20
 description Finance Department
 encapsulation dot1Q 20
 ip address 192.168.7.193 255.255.255.224
 no shutdown

! Interface ke Switch Legal (VLAN 30)
interface FastEthernet1/0.30
 description Legal Department
 encapsulation dot1Q 30
 ip address 192.168.7.225 255.255.255.224
 no shutdown

! Enable parent interface
interface FastEthernet1/0
 description Trunk to Switches Lantai 4
 no shutdown

!============================================
! EIGRP Configuration
!============================================

router eigrp 100
 network 192.168.10.128 0.0.0.3
 network 192.168.7.128 0.0.0.63
 network 192.168.7.192 0.0.0.31
 network 192.168.7.224 0.0.0.31
 passive-interface FastEthernet1/0.10
 passive-interface FastEthernet1/0.20
 passive-interface FastEthernet1/0.30
 no auto-summary

!============================================
! DHCP Pools
!============================================

! Customer Service DHCP
ip dhcp excluded-address 192.168.7.129 192.168.7.136
ip dhcp pool CS-Pool
 network 192.168.7.128 255.255.255.192
 default-router 192.168.7.129
 dns-server 8.8.8.8

! Keuangan DHCP
ip dhcp excluded-address 192.168.7.193 192.168.7.199
ip dhcp pool Keuangan-Pool
 network 192.168.7.192 255.255.255.224
 default-router 192.168.7.193
 dns-server 8.8.8.8

! Legal DHCP
ip dhcp excluded-address 192.168.7.225 192.168.7.231
ip dhcp pool Legal-Pool
 network 192.168.7.224 255.255.255.224
 default-router 192.168.7.225
 dns-server 8.8.8.8

!============================================
! Basic Configuration
!============================================

no ip domain-lookup
service password-encryption

line con 0
 password cisco
 login
 logging synchronous

line vty 0 4
 password cisco
 login

enable secret class

banner motd #
========================================
 ROUTER LANTAI 4 - ARA Tech
========================================
#

end
write memory
