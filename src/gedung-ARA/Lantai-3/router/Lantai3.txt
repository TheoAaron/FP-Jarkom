!============================================
! ROUTER LANTAI 3 - Gedung ARA Tech
! Departemen: R&D, People Development
! Network Aggregation: 192.168.16.0/24
!============================================

enable
configure terminal

hostname Router-Lantai3

! Interface ke Router Gedung ARA
interface FastEthernet0/0
 description Connection to Router-Gedung-ARA
 ip address 192.168.20.10 255.255.255.252
 no shutdown

! Interface ke Switch R&D (VLAN 10)
interface FastEthernet1/0.10
 description Research & Development
 encapsulation dot1Q 10
 ip address 192.168.16.1 255.255.255.128
 no shutdown

! Interface ke Switch People Development (VLAN 20)
interface FastEthernet1/0.20
 description People Development
 encapsulation dot1Q 20
 ip address 192.168.17.225 255.255.255.224
 no shutdown

! Enable parent interface
interface FastEthernet1/0
 description Trunk to Switches Lantai 3
 no shutdown

!============================================
! EIGRP Configuration
!============================================

router eigrp 100
 network 192.168.20.8 0.0.0.3
 network 192.168.16.0 0.0.0.127
 network 192.168.17.224 0.0.0.31
 passive-interface FastEthernet1/0.10
 passive-interface FastEthernet1/0.20
 no auto-summary

!============================================
! DHCP Pools
!============================================

! R&D DHCP
ip dhcp excluded-address 192.168.16.1 192.168.16.10
ip dhcp pool RnD-Pool
 network 192.168.16.0 255.255.255.128
 default-router 192.168.16.1
 dns-server 8.8.8.8

! People Development DHCP
ip dhcp excluded-address 192.168.17.225 192.168.17.230
ip dhcp pool PeopleDev-Pool
 network 192.168.17.224 255.255.255.224
 default-router 192.168.17.225
 dns-server 8.8.8.8

!============================================
! Basic Configuration
!============================================

no ip domain-lookup
service password-encryption

line con 0
 password cisco
 login
 logging synchronous

line vty 0 4
 password cisco
 login

enable secret class

banner motd #
========================================
 ROUTER LANTAI 3 - ARA Tech
========================================
#

end
write memory
